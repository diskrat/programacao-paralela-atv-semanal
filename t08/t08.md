# Tarefa 8: Coerência de cache e Falso compartilhamento
Implemente estimativa estocástica de π usando `rand()` para gerar os pontos. Cada thread deve usar uma variável privada para contar os acertos e acumular o total em uma variável global com `#pragma omp critical`. Depois, implemente uma segunda versão em que cada thread escreve seus acertos em uma posição distinta de um vetor compartilhado. A acumulação deve ser feita em um laço serial após a região paralela. Compare o tempo de execução das duas versões. Em seguida, substitua `rand()` por `rand_r()` em ambas e compare novamente. Explique o comportamento dos quatro programas com base na coerência de cache e nos efeitos do falso compartilhamento.

## Resultado obtido

Os tempos médios foram:

- `acumulador + rand()`: `pi ~= 3.141280400000`, `1904.111726 ms`
- `vetor + rand()`: `pi ~= 3.141870800000`, `1959.889469 ms`
- `acumulador + rand_r()`: `pi ~= 3.099159600000`, `196.665760 ms`
- `vetor + rand_r()`: `pi ~= 3.113093200000`, `291.736417 ms`


## Análise

```c
double count_accumulator(long shots,rand_func rand_f,unsigned int *seed){
    long count_global = 0;
    long count = 0;
    #pragma omp parallel default(none) shared(shots,count_global,rand_f) firstprivate(count,seed)
    {
            
        #pragma omp for 
        for (long i = 0; i < shots; i++) {
            double x = (double)rand_f(seed) / RAND_MAX;
            double y = (double)rand_f(seed) / RAND_MAX;
            if (x * x + y * y <= 1.0) {
                {
                    count++;
                }
            }
        }
        #pragma omp critical
        count_global+= count;
    }
    return 4.0 * (double)count_global / (double)shots;
}
```
Usando acumulador com `rand()` apresentou o melhor desempenho, pois a função `rand()` é thread-safe e não sofre de falso compartilhamento.
```c
double count_vector(long shots,rand_func rand_f,unsigned int *seed){
    int max_threads = omp_get_max_threads();
    long *arr = calloc(max_threads,sizeof(long));
    long count_global = 0;
    #pragma omp parallel default(none) shared(shots,count_global,arr,rand_f) firstprivate(seed)
    {
            
        #pragma omp for 
        for (long i = 0; i < shots; i++) {
            double x = (double)rand_f(seed) / RAND_MAX;
            double y = (double)rand_f(seed) / RAND_MAX;
            if (x * x + y * y <= 1.0) {
                {
                    arr[omp_get_thread_num()]++;
                }
            }
        }
    }
    for (int i = 0; i < max_threads;i++) count_global+=arr[i];
    free(arr);
    return 4.0 * (double)count_global / (double)shots;
}
```

Usando vetor com `rand()` apresentou um desempenho ligeiramente pior devido ao acesso concorrente à função `rand()`, que pode causar contenção atrasos, e invalidação de cache devido ao falso compartilhamento, já que múltiplas threads estão acessando o mesmo vetor compartilhado.

Usando acumulador com `rand_r()` apresentou um desempenho significativamente melhor, pois cada thread tem sua própria semente e não há contenção de acesso à função `rand()`, mas uma condição de corrida é inserida toda vez que a função `rand_r()` é chamada. fazendo o valor de π ser impreciso.

Usando vetor com `rand_r()` apresentou um desempenho pior do que o acumulador, pelos mesmos motivos do vetor com `rand()`, a imprecisão de π é melhor do que o acumulador, mas ainda assim é afetada pela condição de corrida na função `rand_r()`. O falso compartilhamento também pode contribuir para a degradação do desempenho e causando cache misses.

# Implementação Completa

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <omp.h>

#define N 1e7
#define RUNS 1
typedef int (*rand_func)(unsigned int*);


int rand_wrapper(unsigned int *seed){
    (void)seed;
    return rand();
}
int rand_r_wrapper(unsigned int *seed){
    return rand_r(seed);
}

double count_accumulator(long shots,rand_func rand_f,unsigned int *seed){
    long count_global = 0;
    long count = 0;
    #pragma omp parallel default(none) shared(shots,count_global,rand_f) firstprivate(count,seed)
    {
            
        #pragma omp for 
        for (long i = 0; i < shots; i++) {
            double x = (double)rand_f(seed) / RAND_MAX;
            double y = (double)rand_f(seed) / RAND_MAX;
            if (x * x + y * y <= 1.0) {
                {
                    count++;
                }
            }
        }
        #pragma omp critical
        count_global+= count;
    }
    return 4.0 * (double)count_global / (double)shots;
}

double count_vector(long shots,rand_func rand_f,unsigned int *seed){
    int max_threads = omp_get_max_threads();
    long *arr = calloc(max_threads,sizeof(long));
    long count_global = 0;
    #pragma omp parallel default(none) shared(shots,count_global,arr,rand_f) firstprivate(seed) 
    {
            
        #pragma omp for 
        for (long i = 0; i < shots; i++) {
            double x = (double)rand_f(seed) / RAND_MAX;
            double y = (double)rand_f(seed) / RAND_MAX;
            if (x * x + y * y <= 1.0) {
                {
                    arr[omp_get_thread_num()]++;
                }
            }
        }
    }
    for (int i = 0; i < max_threads;i++) count_global+=arr[i];
    free(arr);
    return 4.0 * (double)count_global / (double)shots;
}

double elapsed_ms(struct timespec start, struct timespec end){
    return (end.tv_sec - start.tv_sec) * 1e3 + (end.tv_nsec - start.tv_nsec) / 1e6;
}

double compare_time(long shots, double (*func)(long, rand_func, unsigned int*), rand_func rand_f, unsigned int *seed, double *pi_estimate){
    struct timespec start, end;
    double time_sum = 0.0;

    for (int r = 0; r < RUNS; r++) {
        clock_gettime(CLOCK_MONOTONIC, &start);
        *pi_estimate = func(shots, rand_f, seed);
        clock_gettime(CLOCK_MONOTONIC, &end);
        time_sum += elapsed_ms(start, end);
    }

    return time_sum / RUNS;
}
int main(void) {
    unsigned int seed = 42;
    double pi_estimate = 0.0;

    double time_sum = compare_time(N, count_accumulator, rand_wrapper, &seed, &pi_estimate);
    printf("[acumulador + rand()]   pi ~= %.12f, tempo medio = %f ms (%d execucoes)\n",
            pi_estimate, time_sum, RUNS);

    time_sum = compare_time(N, count_vector, rand_wrapper, &seed, &pi_estimate);
    printf("[vetor + rand()]        pi ~= %.12f, tempo medio = %f ms (%d execucoes)\n",
            pi_estimate, time_sum, RUNS);

    time_sum = compare_time(N, count_accumulator, rand_r_wrapper, &seed, &pi_estimate);
    printf("[acumulador + rand_r()] pi ~= %.12f, tempo medio = %f ms (%d execucoes)\n",
            pi_estimate, time_sum, RUNS);

    time_sum = compare_time(N, count_vector, rand_r_wrapper, &seed, &pi_estimate);
    printf("[vetor + rand_r()]      pi ~= %.12f, tempo medio = %f ms (%d execucoes)\n",
            pi_estimate, time_sum, RUNS);
    
    return 0;
}
```
